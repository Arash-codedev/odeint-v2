// In header: <boost/numeric/odeint/util/unit_helper.hpp>

template<typename Unit, typename Value> 
struct inverse_time<boost::units::quantity< Unit, Value >> {
  // types
  typedef boost::units::quantity< Unit, Value >                                    time_type;    
  typedef boost::units::get_dimension< time_type >::type                           dimension;    
  typedef boost::units::get_system< time_type >::type                              system;       
  typedef boost::mpl::divides< boost::units::dimensionless_type, dimension >::type inv_dimension;
  typedef boost::units::unit< inv_dimension, system >                              inv_unit;     
  typedef boost::units::quantity< inv_unit, Value >                                type;         
};
