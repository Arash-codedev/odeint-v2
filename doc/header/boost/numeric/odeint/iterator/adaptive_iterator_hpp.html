Header <boost/numeric/odeint/iterator/adaptive_iterator.hpp>

namespace boost {
  namespace numeric {
    namespace odeint {
      template<typename Stepper, typename System> 
        class adaptive_iterator<Stepper, System, controlled_stepper_tag>;
      template<typename Stepper, typename System> 
        class adaptive_iterator<Stepper, System, dense_output_stepper_tag>;
      template<typename Stepper, typename System> 
        adaptive_iterator< Stepper, System > 
        make_adaptive_iterator_begin(Stepper stepper, System system, 
                                     typename Stepper::state_type & x, 
                                     typename Stepper::time_type t, 
                                     typename Stepper::time_type t_end, 
                                     typename Stepper::time_type dt);
      template<typename Stepper, typename System> 
        adaptive_iterator< Stepper, System > 
        make_adaptive_iterator_end(Stepper stepper, System system, 
                                   typename Stepper::state_type & x);
      template<typename Stepper, typename System> 
        std::pair< adaptive_iterator< Stepper, System >, adaptive_iterator< Stepper, System > > 
        make_adaptive_range(Stepper stepper, System system, 
                            typename Stepper::state_type & x, 
                            typename Stepper::time_type t_start, 
                            typename Stepper::time_type t_end, 
                            typename Stepper::time_type dt);
    }
  }
}
