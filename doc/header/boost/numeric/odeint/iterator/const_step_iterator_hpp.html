Header <boost/numeric/odeint/iterator/const_step_iterator.hpp>

template<typename Stepper, typename System, 
         typename StepperTag = typename base_tag< typename Stepper::stepper_category >::type> 
  class const_step_iterator;namespace boost {
  namespace numeric {
    namespace odeint {
      template<typename Stepper, typename System> 
        class const_step_iterator<Stepper, System, stepper_tag>;
      template<typename Stepper, typename System> 
        class const_step_iterator<Stepper, System, dense_output_stepper_tag>;
      template<typename Stepper, typename System> 
        const_step_iterator< Stepper, System > 
        make_const_step_iterator_begin(Stepper stepper, System system, 
                                       typename Stepper::state_type & x, 
                                       typename Stepper::time_type t, 
                                       typename Stepper::time_type t_end, 
                                       typename Stepper::time_type dt);
      template<typename Stepper, typename System> 
        const_step_iterator< Stepper, System > 
        make_const_step_iterator_end(Stepper stepper, System system, 
                                     typename Stepper::state_type & x);
      template<typename Stepper, typename System> 
        std::pair< const_step_iterator< Stepper, System >, const_step_iterator< Stepper, System > > 
        make_const_step_range(Stepper stepper, System system, 
                              typename Stepper::state_type & x, 
                              typename Stepper::time_type t_start, 
                              typename Stepper::time_type t_end, 
                              typename Stepper::time_type dt);
    }
  }
}
